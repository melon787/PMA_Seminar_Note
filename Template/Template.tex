\section{Sequences}

\begin{definition}
A sequence $(x_n)$ in $\R$ converges to $L \in \R$ if for every $\varepsilon > 0$,
there exists $N \in \N$ such that
\[
    n \geq N \implies |x_n - L| < \varepsilon.
\]
\end{definition}

\begin{theorem}
Every convergent sequence is bounded.
\end{theorem}

\begin{proof}
Suppose that $(x_n)$ converges to $L$. Taking $\varepsilon = 1$, there exists
$N \in \N$ such that
\[
    n \geq N \implies |x_n - L| < 1.
\]
Hence, for all $n \geq N$,
\[
    |x_n| \leq |x_n - L| + |L| < 1 + |L|.
\]
The remaining terms $x_1, \dots, x_{N-1}$ are finitely many, so they are bounded.
Therefore, $(x_n)$ is bounded.
\end{proof}

\begin{remark}
The main idea is that convergence controls the tail of the sequence, while the
initial finitely many terms are automatically bounded.
\end{remark}
